local _themeListeners = {}

-- ลงทะเบียน callback รับ theme ใหม่เมื่อเปลี่ยน
local function onThemeChange(fn)
    table.insert(_themeListeners, fn)
end

-- เรียก callback ทั้งหมด (ใน pcall กัน error ถ้า element destroy ไปแล้ว)
local function fireThemeChange()
    for _, fn in ipairs(_themeListeners) do
        pcall(fn, TH)
    end
end

-- ════════════════════════════════════════════════════════════
--  🔧 Color Utilities
-- ════════════════════════════════════════════════════════════
local function hexToColor3(hex)
    hex = hex:gsub("^#", "")
    if #hex ~= 6 then return nil end
    local r = tonumber(hex:sub(1,2), 16)
    local g = tonumber(hex:sub(3,4), 16)
    local b = tonumber(hex:sub(5,6), 16)
    if not (r and g and b) then return nil end
    return Color3.fromRGB(r, g, b)
end

local function color3ToHex(c)
    return string.format("%02X%02X%02X",
        math.round(c.R * 255),
        math.round(c.G * 255),
        math.round(c.B * 255))
end

local function rgbStringToColor3(s)
    local r,g,b = s:match("(%d+)%s*,%s*(%d+)%s*,%s*(%d+)")
    r,g,b = tonumber(r), tonumber(g), tonumber(b)
    if not (r and g and b) then return nil end
    return Color3.fromRGB(math.clamp(r,0,255), math.clamp(g,0,255), math.clamp(b,0,255))
end

local function color3ToRgbString(c)
    return math.round(c.R*255)..","..math.round(c.G*255)..","..math.round(c.B*255)
end

local function hsvToColor3(h, s, v) return Color3.fromHSV(h, s, v) end
local function color3ToHsv(c) return Color3.toHSV(c) end

-- ใช้ overlay ระดับ Win แทน page-level เพื่อ fade ทุกอย่าง
local _winOverlay = nil  -- สร้างครั้งเดียว ใช้ซ้ำ

local function getWinOverlay(win)
    if _winOverlay and _winOverlay.Parent then return _winOverlay end
    -- parent คือ ScreenGui (ui) เพื่อหลีกเลี่ยง ClipsDescendants ของ Win
    local ui = win and win.Parent
    local ov = Instance.new("Frame")
    ov.Name                   = "__WinFadeOverlay"
    ov.Parent                 = ui or win
    ov.BackgroundColor3       = TH.bg
    ov.BorderSizePixel        = 0
    -- ขนาด+ตำแหน่งตาม Win
    ov.Position               = win and win.Position or UDim2.new(0,0,0,0)
    ov.Size                   = win and win.Size     or UDim2.new(1,0,1,0)
    ov.ZIndex                 = 9999
    ov.BackgroundTransparency = 1
    ov.Active                 = false
    -- ติดตามขนาด/ตำแหน่ง Win
    if win then
        win:GetPropertyChangedSignal("Position"):Connect(function() ov.Position = win.Position end)
        win:GetPropertyChangedSignal("Size"):Connect(function()     ov.Size     = win.Size     end)
    end
    local _ovc = Instance.new("UICorner"); _ovc.CornerRadius = UDim.new(0,6); _ovc.Parent = ov
    _winOverlay = ov
    return ov
end

local function pageOut(pageFrame, done, win)
    local ov = getWinOverlay(win or (pageFrame and pageFrame:FindFirstAncestorOfClass("Frame")))
    if not ov then if done then done() end return end
    ov.BackgroundColor3 = TH.bg
    ov.BackgroundTransparency = 1
    ov.Active = true
    tw(ov, {BackgroundTransparency = 0}, 0.12, Enum.EasingStyle.Sine)
    task.delay(0.13, function()
        if done then done() end
    end)
end

local function pageIn(pageFrame, win)
    local ov = _winOverlay
    if not ov or not ov.Parent then return end
    task.delay(0.04, function()
        tw(ov, {BackgroundTransparency = 1}, 0.20, Enum.EasingStyle.Sine)
        task.delay(0.21, function()
            if ov and ov.Parent then ov.Active = false end
        end)
    end)
end

            return

        local oldPage = currentPage
        currentPage   = newPage
        currentIdx    = newIdx

        if oldPage then
            pageOut(oldPage, function()
                UIPageLayout:JumpToIndex(newIdx - 1)
                task.delay(0.02, function()
                    pageIn(newPage, Win)
                    transitioning = false
                end)
            end, Win)
        else
            UIPageLayout:JumpToIndex(newIdx - 1)
            pageIn(newPage, Win)
            transitioning = false
        end
    end

            -- encode: 9 สี × 6 hex = 54 ตัวอักษร
            local function encodeTheme(t)
                local out = t.name and (t.name:gsub("[^%w%-_]","_"):sub(1,16)..":") or ""
                for _, k in ipairs(SLOT_KEYS) do
                    out = out .. color3ToHex(t[k])
                end
                return out
            end

            local function decodeTheme(code, fallbackName)
                -- strip whitespace
                code = code:gsub("%s","")
                -- optional name prefix: "Name:HEXHEX..."
                local name, hex = code:match("^([^:]+):([%x]+)$")
                if not name then
                    name = fallbackName or "Imported"
                    hex  = code:gsub("[^%x]",""):upper()
                else
                    hex = hex:upper()
                end
                if #hex ~= 54 then return nil end
                local t = { name = name }
                for i, k in ipairs(SLOT_KEYS) do
                    local s = (i-1)*6 + 1
                    local c = hexToColor3(hex:sub(s,s+5))
                    if not c then return nil end
                    t[k] = c
                end
                return t
            end

            ImportLbl.Text = "วางโค้ดจากเพื่อน — กด โหลด เพื่อใช้"
            ImportLbl.TextColor3 = TH.su; ImportLbl.TextSize = 9
            ImportLbl.TextXAlignment = Enum.TextXAlignment.Left
            onThemeChange(function(t) tw(ImportLbl,{TextColor3=t.su},0.22) end)

            local ImportWrap = Instance.new("Frame")
            ImportWrap.Parent = Page; ImportWrap.BackgroundTransparency = 1
            ImportWrap.BorderSizePixel = 0; ImportWrap.Size = UDim2.new(1,-28,0,30)
            local ImportLayout = Instance.new("UIListLayout")
            ImportLayout.Parent = ImportWrap; ImportLayout.FillDirection = Enum.FillDirection.Horizontal
            ImportLayout.SortOrder = Enum.SortOrder.LayoutOrder; ImportLayout.Padding = UDim.new(0,6)
            ImportLayout.VerticalAlignment = Enum.VerticalAlignment.Center

            local ImportBox = Instance.new("TextBox")
            ImportBox.Parent = ImportWrap; ImportBox.BackgroundColor3 = TH.cd
            ImportBox.BorderSizePixel = 0; ImportBox.Size = UDim2.new(1,-74,1,0)
            ImportBox.LayoutOrder = 1; ImportBox.Font = Enum.Font.GothamBold
            ImportBox.Text = ""; ImportBox.TextColor3 = WT
            ImportBox.TextSize = 9; ImportBox.PlaceholderText = "วางโค้ดที่นี่..."
            ImportBox.ClearTextOnFocus = false
            local _imbc = Instance.new("UICorner"); _imbc.CornerRadius = UDim.new(0,6); _imbc.Parent = ImportBox
            local ImportStroke = Instance.new("UIStroke"); ImportStroke.Parent=ImportBox; ImportStroke.Color=TH.br; ImportStroke.Thickness=1
            onThemeChange(function(t) tw(ImportBox,{BackgroundColor3=t.cd},0.22); ImportStroke.Color=t.br end)

            local LoadBtn = Instance.new("TextButton")
            LoadBtn.Parent = ImportWrap; LoadBtn.BackgroundColor3 = TH.dk
            LoadBtn.BorderSizePixel = 0; LoadBtn.Size = UDim2.new(0,68,1,0)
            LoadBtn.LayoutOrder = 2; LoadBtn.AutoButtonColor = false
            LoadBtn.Font = Enum.Font.GothamBold; LoadBtn.Text = "โหลด"
            LoadBtn.TextColor3 = TH.ac; LoadBtn.TextSize = 11
            local _ldbc = Instance.new("UICorner"); _ldbc.CornerRadius = UDim.new(0,6); _ldbc.Parent = LoadBtn

            local function flashImport(ok)
                tw(ImportStroke, {Color = ok and TH.ac or Color3.fromRGB(240,60,60)}, 0.1)
                task.delay(1, function() ImportStroke.Color = TH.br end)
            end

            local function applyDecodedTheme(decoded)
                for k, v in pairs(decoded) do editTheme[k] = v end
                -- อัปเดต row swatches + chips + picker
                for i, def in ipairs(colorDefs) do
                    local nc = editTheme[def.key]
                    if rowEls[i] then
                        tw(rowEls[i].swatch, {BackgroundColor3=nc}, 0.15)
                        rowEls[i].hexLbl.Text = "#"..color3ToHex(nc)
                    end
                    if previewChips[i] then
                        tw(previewChips[i].chip, {BackgroundColor3=nc}, 0.15)
                    end
                end
                ExportBox.Text = encodeTheme(editTheme)
                -- reset picker ถ้าเปิดอยู่
                if selIdx >= 1 then
                    local key = colorDefs[selIdx].key
                    ph, ps, pv = color3ToHsv(editTheme[key])
                    syncPicker()
                end
            end

            LoadBtn.MouseButton1Down:Connect(function() tw(LoadBtn,{Size=UDim2.new(0,64,1,0)},0.08,Enum.EasingStyle.Sine) end)
            LoadBtn.MouseButton1Click:Connect(function()
                tw(LoadBtn,{Size=UDim2.new(0,68,1,0)},0.12,Enum.EasingStyle.Back)
                local code = ImportBox.Text
                local importName = NameBox.Text ~= "" and NameBox.Text or nil
                local decoded = decodeTheme(code, importName)
                if decoded then
                    applyDecodedTheme(decoded)
                    -- ถ้ามีชื่อใน code ให้เติมใน NameBox
                    if decoded.name and decoded.name ~= "Imported" and NameBox.Text == "" then
                        NameBox.Text = decoded.name
                    end
                    flashImport(true)
                    ImportBox.Text = ""
                    LoadBtn.Text = "✓ โหลดแล้ว"
                    tw(LoadBtn,{BackgroundColor3=TH.ac,TextColor3=Color3.new(1,1,1)},0.1)
                    task.delay(1.5, function()
                        LoadBtn.Text = "โหลด"
                        tw(LoadBtn,{BackgroundColor3=TH.dk,TextColor3=TH.ac},0.2)
                    end)
                else
                    flashImport(false)
                    LoadBtn.Text = "✗ โค้ดผิด"
                    tw(LoadBtn,{BackgroundColor3=Color3.fromRGB(200,50,50),TextColor3=Color3.new(1,1,1)},0.1)
                    task.delay(1.5, function()
                        LoadBtn.Text = "โหลด"
                        tw(LoadBtn,{BackgroundColor3=TH.dk,TextColor3=TH.ac},0.2)
                    end)
                end
            end)
            LoadBtn.MouseEnter:Connect(function() tw(LoadBtn,{BackgroundColor3=TH.ac,TextColor3=Color3.new(1,1,1)},0.14) end)
            LoadBtn.MouseLeave:Connect(function() tw(LoadBtn,{BackgroundColor3=TH.dk,TextColor3=TH.ac},0.14) end)
            onThemeChange(function(t) tw(LoadBtn,{BackgroundColor3=t.dk,TextColor3=t.ac},0.22) end)

            -- sync ExportBox เมื่อแก้สีใน picker
            local origSync = syncPicker
            syncPicker = function()
                origSync()
                ExportBox.Text = encodeTheme(editTheme)
            end
        end

        return TabFunctions
    end

    local WindowObj = {}
    function WindowObj:Tab(...)
        return Tabs:Tab(...)
    end
    return WindowObj
end
return Window
